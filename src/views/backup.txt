<script setup>
import { ref, onMounted } from "vue";

const students = ref([]);

onMounted(async () => {
  try {
    const res = await fetch("http://127.0.0.1:5000/students");
    students.value = await res.json();
  } catch (error) {
    console.error("Error fetching students:", error);
  }
});
</script>

<template>
    <div>
    <div class="card">
        <div class="class-header">
            <h4>Students</h4>
        </div>
            <div class="card-body">
                <table class="table">
                    <thead>
                        <tr>
                            <th>Id</th>
                            <th>name</th>
                            <th>course</th>
                        </tr>
                    </thead>
                    <tbody>
                        <tr v-for="student in students" :key="student.id">
                            <td>{{ student.id }}</td>
                            <td>{{ student.name }}</td>
                            <td>{{ student.course }}</td>
                        </tr>
                    </tbody>
                </table>
            </div>
    </div>
    </div>
</template>
